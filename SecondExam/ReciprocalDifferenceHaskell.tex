> rho :: [Ratio Int] -> Int -> Int -> Ratio Int
> rho fs 0 i = fs !! i
> rho fs n _ 
>   | n < 0 = 0
> rho fs n i = (n*den)%num + rho fs (n-2) (i+1)
>   where
>     num  = numerator next
>     den  = denominator next
>     next = (rho fs (n-1) (i+1)) - (rho fs (n-1) i)

> a fs 0 = fs !! 0
> a fs n = rho fs n 0 - rho fs (n-2) 0
\end{verbatim}
